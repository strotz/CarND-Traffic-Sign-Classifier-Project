from tensorflow.contrib.layers import flatten

def make_random(shape, name):
    # Hyperparameters
    mu = 0
    sigma = 0.1
    return tf.Variable(tf.truncated_normal(shape=shape, mean = mu, stddev = sigma), name=name)

# filter=[filter_width, filter_height, input_depth, output_depth]
def make_convolution_layer(name, input, filter, strides, padding):
    output_depth = filter[3] 
    
    weights=make_random(filter, name='weights_' + name)
    bias=tf.Variable(tf.zeros(output_depth), name='bias_' + name)

    conv = tf.nn.conv2d(input, weights, strides=strides, padding=padding)
    return tf.nn.bias_add(conv, bias)

def make_dropout_layer(input):
    return tf.nn.dropout(input, keep_prob=drop)

def make_pooling_layer(input, k=2, padding='SAME'):
    return tf.nn.max_pool(input, ksize=[1, k, k, 1], strides=[1, k, k, 1], padding=padding)

def make_activation_layer(input):
    return tf.nn.relu(input) 
    
def make_fully_connected(name, input, input_size, output_size):
    weights=make_random([input_size, output_size], name='weights_' + name)
    bias=tf.Variable(tf.zeros(output_size), name='bias_' + name)
    
    logits = tf.matmul(input, weights)
    return tf.nn.bias_add(logits, bias)


def LeNet2(x): # x = tf.placeholder(tf.float32, (None, 32, 32, 3))
    
    # Layer 1: Input = 32x32x3. 
    layer1_ = make_convolution_layer('layer_1', x, [5,5,3,6],[1,1,1,1],'VALID') # Convolutional. Output = 28x28x6.
    layer1_ = make_activation_layer(layer1_) # Activation.
    layer1_ = make_dropout_layer(layer1_) # dropout
    layer1_ = make_pooling_layer(layer1_) # Pooling. Input = 28x28x6. Output = 14x14x6.

    # Layer 2: 
    layer2_ = make_convolution_layer('layer_2', layer1_, [5,5,6,16],[1,1,1,1],'VALID') # Convolutional. Output = 10x10x16.
    layer2_ = make_activation_layer(layer2_) # Activation.
    layer2_ = make_pooling_layer(layer2_) # Pooling. Input = 10x10x16. Output = 5x5x16.

    # Flatten. Input = 5x5x16. Output = 400.
    layer2_ = tf.reshape(layer2_, [-1,400])

    # Layer 3:
    layer3_ = make_fully_connected('layer_3', layer2_, 400, 120) # Fully Connected. Input = 400. Output = 100
    layer3_ = make_dropout_layer(layer3_) # dropout
    layer3_ = make_activation_layer(layer3_) # Activation.

    # Layer 4:
    layer4_ = make_fully_connected('layer_4', layer3_, 120, 80) # Fully Connected. Input = 120. Output = 80.
    layer4_ = make_activation_layer(layer4_) # Activation.

    # Layer 5: 
    logits = make_fully_connected('layer_5', layer4_, 80, n_classes) # Fully Connected. Input = 84. Output = 10.
    
    return logits

Question 3¶

What does your final architecture look like? (Type of model, layers, sizes, connectivity, etc.) For reference on how to build a deep neural network using TensorFlow, see Deep Neural Network in TensorFlow from the classroom.

Answer:

Current architecture is classic LeNet with addition of 2 dropout layers.

First convolution layer has relu has 6 output features, relu activation and dropout with variable rate. Output of this layer passed via max pooling with kernel size 2 and stride 2. This effectivly reduces number of signals from 32x32x3 to 14x14x6

Next layer also convolution with max pooling. After multiple experiments I have not observed any improvments if I would add dropout. The size of this layer output is 5x5x16. I also experimented with number of channels for this convolution and 16 is quite reasonable. Increase of features contributes to over-fitting.

Next 3 layers are fully connected with relu activation. Each next layer hes less neurons than previous: 120 => 80 => 43

I was surpised, but adding dropout to first fully connectyed layer reduced difference between trainong and validation estimation, i.e. contibutes to better generalization. I have not observed such difference when I've enabled dropout for convolutional layers.

Last layer does not have explicit activation, yet future code implies that it will use softmax (training algorith reduces softmax cross entropy)

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y, drop: 1.0})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

Answer:

I'm using TensorFlow implementation of Adam optimizer that has next advantages to simple gradient descend optimizer:

  • it reduces learning rate with epichs
  • uses momentum

No L1 or L2 regularization were applied After addition of dropout to layer 1 and 3, I observe smaller difference between accuracy estimation on training and validation data sets.

Dropout keep probability increased every epich starting from 0.5 to 1.0

From picture above it is clear that there is a very little improvments of validation accuracy starting from epoch 25.

I've tried to train network using differnt sizes of batch. Seems like large batches make gradient and learning more "smooth". i.e. with smaller batches I've observed negative improvments of accuracy of validation set. So, batch size 512 worked well.
